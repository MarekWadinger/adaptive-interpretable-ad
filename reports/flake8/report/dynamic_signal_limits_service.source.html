87 def preprocess(
88 x,
89 col):
90 """Preprocess the input data.
91  
92 Args:
93 x (Union[pd.Series, tuple, dict, MQTTMessage, bytes]): The input data
94 to be preprocessed.
95 col (Union[str, List[str]]): The column(s) to be extracted from the
96 input data.
97  
98 Returns:
99 dict: The preprocessed data.
100  
101 Examples:
102 >>> series = pd.Series([1.], name=pd.to_datetime('2023-01-01'),
103 ... index=["sensor_1"])
104 >>> preprocess(series, 'sensor_1')
105 {'time': Timestamp('2023-01-01 00:00:00'), 'data': {'sensor_1': 1.0}}
106  
107 >>> series_tuple = (pd.to_datetime('2023-01-01'), series)
108 >>> preprocess(series_tuple, 'sensor_1')
109 {'time': Timestamp('2023-01-01 00:00:00'), 'data': {'sensor_1': 1.0}}
110  
111 >>> data_dict = {'time': pd.to_datetime('2023-01-01'), 'sensor_1': 1.}
112 >>> out = preprocess(data_dict, ['sensor_1'])
113 >>> out.keys(), out['data'].keys()
114 (dict_keys(['time', 'data']), dict_keys(['sensor_1']))
115  
116 >>> mqtt_message = MQTTMessage()
117 >>> mqtt_message.timestamp = 1672527600.0
118 >>> mqtt_message.payload = b'1.'
119 >>> mqtt_message.topic = b'sensors/sensor_1'
120 >>> out = preprocess(mqtt_message, '1')
121 >>> out.keys(), out['data'].keys()
122 (dict_keys(['time', 'data']), dict_keys(['sensor_1']))
123  
124 >>> binary_data = b'1.0'
125 >>> out = preprocess(binary_data, 'sensor_1')
126 >>> out.keys(), out['data'].keys()
127 (dict_keys(['time', 'data']), dict_keys(['sensor_1']))
128 """
129 if isinstance(x, pd.Series):
130 col = [col] if not isinstance(col, list) else col
131 return {"time": x.name.tz_localize(None),
132 "data": x[col].to_dict()
133 }
134 elif isinstance(x, tuple) and isinstance(x[1], (pd.Series)):
135 col = [col] if not isinstance(col, list) else col
136 return {"time": x[0].tz_localize(None),
137 "data": x[1][col].to_dict()
138 }
139 elif isinstance(x, dict):
140 return {"time": dt.datetime.now().replace(microsecond=0),
141 "data": {k: v for k, v in x.items() if k in col}
142 }
143 elif isinstance(x, MQTTMessage):
144 return {"time": dt.datetime
145 .fromtimestamp(x.timestamp).replace(microsecond=0),
146 "data": {x.topic.split("/")[-1]: float(x.payload)}
147 }
148 elif isinstance(x, bytes):
149 return {"time": dt.datetime.now().replace(microsecond=0),
150 "data": {col: float(x.decode("utf-8"))}
151 }

253 The function returns a data source stream object based on the
254 configuration settings.
255 If the 'path' key is present in the config, it returns a stream from an
256 iterable of

258 stream from

260 present,

262 of the expected keys are found, it raises a RuntimeError.
263  
264 Args:
265 config (dict): The configuration dictionary.
266 topic (str): The topic to subscribe to for MQTT or Kafka sources.
267 debug (bool, optional): Enable debug mode. Defaults to False.
268  
269 Returns:
270 stream.Stream: The data source stream object.
271  
272 Raises:
273 RuntimeError: If the data format is incorrect.
274  
275 Examples:
276 >>> config = {
277 ... "path": "path/to/input/data.csv",
278 ... "data": pd.DataFrame({"A": [1, 2, 3], "B": [4, 5, 6]})}
279 >>> topic = "test"
